# AudioRelay command(For pulseaudio or pipewire)
#Load
pactl load-module module-simple-protocol-tcp source=alsa_output.pci-0000_00_1b.analog-stereo.monitor record=true port=8000 rate=44100 format=s16le channels=2

or

pactl load-module module-simple-protocol-tcp source=alsa_output.pci-0000_00_1b.0.analog-stereo.monitor record=true port=8000 rate=44100 format=s16le channels=2

#unload

pactl unload-module module-simple-protocol-tcp

Use Simple protocol player app on your phone
